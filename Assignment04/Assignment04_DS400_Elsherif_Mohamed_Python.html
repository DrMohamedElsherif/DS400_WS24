# Open-Meteo Historical Weather API
url = "https://archive-api.open-meteo.com/v1/era5"

# List to store the weather data for each year
weather_data = []

# Date range from 1940 to 2024
start_year = 1940
end_year = 2024

# Loop through each year 
for year in range(start_year, end_year + 1):
    # Request parameters for the API
    params = {
        'latitude': latitude,
        'longitude': longitude,
        'start_date': f'{year}-01-01',
        'end_date': f'{year}-12-31',
        'hourly': 'temperature_2m',
        'temperature_unit': 'celsius',
        'timezone': 'Europe/Berlin'
    }
    
    # Send request to API
    response = requests.get(url, params=params)
    
    response.status_code == 200
    data = response.json()
    # Extract temperature data for the year
    if 'hourly' in data:
        temperatures = data['hourly']['temperature_2m']
        dates = data['hourly']['time']
            
        # Create a DataFrame for temperatures
        df = pd.DataFrame({'Date': pd.to_datetime(dates),'Temperature': temperatures})
            
        # Add the year to the DataFrame
        df['Year'] = df['Date'].dt.year
            
        # Append the DataFrame for this year to the list
        weather_data.append(df)

# Combine all the DataFrames for the years 1940-2024 into one DataFrame
all_data = pd.concat(weather_data, ignore_index=True)

# GitHub page URL
URL = "https://github.com/topics/machine-learning"

# Fetch page content
response = requests.get(URL)

# Parse the page content
tree = html.fromstring(response.content)

# XPath repository names
repo_names = tree.xpath('//*[contains(concat(" ", @class, " "), " wb-break-word ")]')

# XPath repository descriptions
repo_descriptions = tree.xpath('//*[contains(concat(" ", @class, " "), " color-fg-muted ") and contains(concat(" ", @class, " "), " mb-0 ")]')

# XPath last updated dates
last_updated = tree.xpath('//*[contains(concat(" ", @class, " "), " color-fg-muted ")]//*[contains(concat(" ", @class, " "), " no-wrap ")]')

# XPath for stars numbers
stars = tree.xpath('(//*[contains(concat(" ", @class, " "), " btn ")]//span[not(contains(text(), "Star")) and not(contains(text(), "All")) and not(contains(text(), "Most"))])[position() <= 21]')

# Clean the star count text
stars_text = [star.text_content().strip().split(' ')[0] for star in stars]

# Compile data as dataframe
repositories = []
for name, desc, updated, star in zip(repo_names, repo_descriptions, last_updated, stars_text):
    repositories.append({
        "Repository Name": name.text_content().strip(),
        "Repository URL": f"https://github.com{name.get('href')}",
        "Description": desc.text_content().strip(),
        "Last Updated": updated.text_content().strip(),
        "Stars": star  
    })
df = pd.DataFrame(repositories)

# Ensure 'Stars' are treated as numeric values
def numberize_stars(star_value):
    if 'k' in star_value:
        return float(star_value.replace('k', '').strip()) * 1000
    elif 'M' in star_value:
        return float(star_value.replace('M', '').strip()) * 1000000
    else:
        return float(star_value)

# Numberize the 'Stars' column
df['Stars Numeric'] = df['Stars'].apply(numberize_stars)

# Sort repositories by stars and get top 10
top_10_repositories = df.nlargest(10, 'Stars Numeric')
