1.[R] Program a function that describes a dataset. The function should be called describe_data() and take an R dataframe as input.

describe_data <- function(df) {
  # Get number of rows and columns
  cat("Number of rows and columns:\n")
  cat("Rows:", nrow(df), "\n")
  cat("Columns:", ncol(df), "\n\n")
  
  # Categorical variables (factors)
  # Categorical variables (factors)
  cat("Categorical variables:\n")
  categorical_vars <- sapply(df, is.factor)
  
  for (var in names(df)[categorical_vars]) {
    cat("Variable:", var, "\n")
    cat("------------------------------\n")
    
    # Get the valid categories and NA counts separately
    freqs_non_na <- table(df[[var]], useNA = "no")  # excluding NA
    na_count <- sum(is.na(df[[var]]))  # count NA values
    
    total_non_na <- sum(freqs_non_na)  # total valid values
    
    # Print non-missing categories with percentages excluding NA
    for (category in names(freqs_non_na)) {
      cat(category, ":", freqs_non_na[category], 
          "(", round((freqs_non_na[category] / total_non_na) * 100, 2), "%)\n")
    }
    
    # Print NA as a separate category
    if (na_count > 0) {
      cat("NA:", na_count, "(", round((na_count / (total_non_na + na_count)) * 100, 2), "%)\n")
    }
    
    cat("\n")
  }
  
  
  # Continuous variables (numeric or integer)
  cat("Continuous variables:\n")
  continuous_vars <- sapply(df, is.numeric)
  if (any(continuous_vars)) {
    continuous_summary <- data.frame(
      Variable = character(0),
      N = integer(0),
      min = numeric(0),
      max = numeric(0),
      median = numeric(0),
      mean = numeric(0),
      sd = numeric(0),
      Missings = integer(0)
    )
    
    for (var in names(df)[continuous_vars]) {
      # Calculate statistics for each continuous variable
      N <- sum(!is.na(df[[var]]))
      min_val <- min(df[[var]], na.rm = TRUE)
      max_val <- max(df[[var]], na.rm = TRUE)
      median_val <- median(df[[var]], na.rm = TRUE)
      mean_val <- mean(df[[var]], na.rm = TRUE)
      sd_val <- sd(df[[var]], na.rm = TRUE)
      missing <- sum(is.na(df[[var]]))
      
      # Store statistics for each variable in the summary data frame
      continuous_summary <- rbind(continuous_summary, data.frame(
        Variable = var, N = N, min = min_val, max = max_val, 
        median = median_val, mean = mean_val, sd = sd_val, Missings = missing
      ))
    }
    
    # Print the summary in a clean table format
    print(continuous_summary)
  } else {
    cat("No continuous variables found.\n")
  }
  
  # Other variables (character, logical, etc.)
  cat("\nOther variables:\n")
  other_vars <- sapply(df, function(x) !(is.factor(x) || is.numeric(x)))
  for (var in names(df)[other_vars]) {
    missing_vals <- sum(is.na(df[[var]]))
    cat(var, ":", missing_vals, "NA\n")
  }
}
